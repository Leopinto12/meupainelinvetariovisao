    <script>
        // Inicializa√ß√£o de Datas
        document.getElementById('current-date').innerText = new Date().toLocaleDateString('pt-BR', { weekday: 'long', day: 'numeric', month: 'long' });

        // Banco de Dados Local (LocalStorage)
        let db = JSON.parse(localStorage.getItem('organizador_db')) || { tasks: [], files: [], notes: [] };

        function saveData() {
            localStorage.setItem('organizador_db', JSON.stringify(db));
            renderAll();
        }

        function addItem(type, inputId) {
            const val = document.getElementById(inputId).value;
            if(!val) return;
            db[type].push({ id: Date.now(), text: val, date: new Date().toLocaleString() });
            document.getElementById(inputId).value = '';
            saveData();
        }

        function addFile() {
            const name = document.getElementById('fileName').value;
            const url = document.getElementById('fileUrl').value;
            if(!name || !url) return;
            db.files.push({ id: Date.now(), name, url, date: new Date().toLocaleDateString() });
            document.getElementById('fileName').value = '';
            document.getElementById('fileUrl').value = '';
            saveData();
        }

        function deleteItem(type, id) {
            db[type] = db[type].filter(item => item.id !== id);
            saveData();
        }

        function renderAll() {
            // Render Tarefas
            const taskDiv = document.getElementById('taskList');
            taskDiv.innerHTML = db.tasks.map(t => `
                <div class="list-entry">
                    <div class="entry-info">
                        <strong>${t.text}</strong>
                        <span>Adicionado em: ${t.date}</span>
                    </div>
                    <button onclick="deleteItem('tasks', ${t.id})" style="background:#ef4444; padding:5px 10px;">X</button>
                </div>
            `).join('');

            // Render Arquivos
            const fileDiv = document.getElementById('fileList');
            fileDiv.innerHTML = db.files.map(f => `
                <a href="${f.url}" target="_blank" class="list-entry" style="text-decoration:none">
                    <div class="entry-info">
                        <strong>üìÑ ${f.name}</strong>
                        <span>${f.date} ‚Ä¢ <span class="tag">Storage Externo</span></span>
                    </div>
                </a>
            `).join('');
        }

        window.onload = renderAll;
    </script>
