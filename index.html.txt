    <script>
        // Data e Sauda√ß√£o
        const dateText = document.getElementById('dateText');
        const greeting = document.getElementById('greeting');
        const now = new Date();
        
        const options = { weekday: 'long', day: 'numeric', month: 'long' };
        dateText.innerText = now.toLocaleDateString('pt-BR', options);

        const hour = now.getHours();
        if(hour < 12) greeting.innerText = "Bom dia!";
        else if(hour < 18) greeting.innerText = "Boa tarde!";
        else greeting.innerText = "Boa noite!";

        // L√≥gica de Tarefas
        function addTask() {
            const input = document.getElementById('taskInput');
            if (input.value.trim() === '') return;

            const task = { id: Date.now(), text: input.value, done: false };
            renderTask(task);
            saveTask(task);
            input.value = '';
        }

        function renderTask(task) {
            const li = document.createElement('li');
            li.id = task.id;
            li.innerHTML = `
                <div>
                    <input type="checkbox" ${task.done ? 'checked' : ''} onchange="toggleTask(${task.id})">
                    <span style="${task.done ? 'text-decoration: line-through; color: gray' : ''}">${task.text}</span>
                </div>
                <button class="delete-btn" onclick="deleteTask(${task.id})">Remover</button>
            `;
            document.getElementById('taskList').appendChild(li);
        }

        function saveTask(task) {
            let tasks = JSON.parse(localStorage.getItem('tasks_db') || '[]');
            tasks.push(task);
            localStorage.setItem('tasks_db', JSON.stringify(tasks));
        }

        function loadTasks() {
            let tasks = JSON.parse(localStorage.getItem('tasks_db') || '[]');
            tasks.forEach(renderTask);
        }

        function deleteTask(id) {
            let tasks = JSON.parse(localStorage.getItem('tasks_db') || '[]');
            tasks = tasks.filter(t => t.id !== id);
            localStorage.setItem('tasks_db', JSON.stringify(tasks));
            document.getElementById(id).remove();
        }

        function toggleTask(id) {
            let tasks = JSON.parse(localStorage.getItem('tasks_db') || '[]');
            tasks = tasks.map(t => t.id === id ? {...t, done: !t.done} : t);
            localStorage.setItem('tasks_db', JSON.stringify(tasks));
            location.reload(); // Atualiza para aplicar o risco no texto
        }

        window.onload = loadTasks;
    </script>
